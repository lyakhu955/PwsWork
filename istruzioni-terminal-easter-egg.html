<script>
(function() {
    fetch('https://api.github.com/repos/lyakhu955/MioRepo/commits?per_page=1', {
        headers: { 'Accept': 'application/vnd.github.v3+json' }
    })
    .then(function(res) {
        var link = res.headers.get('Link') || '';
        var match = link.match(/page=(\d+)>;\s*rel="last"/);
        if (match) {
            var count = parseInt(match[1], 10);
            var badge = document.querySelector('.app-version');
            if (badge) badge.textContent = 'v' + count;
            window._devCommitCount = count;
        }
    })
    .catch(function() {});
})();

function openDevModal() {
    var modal = document.getElementById('dev-modal');
    modal.classList.add('active');
    var content = document.getElementById('dev-typewriter-content');
    content.innerHTML = '';
    var i = 0;
    _devTypeTimer = setInterval(function() {
        if (i >= _devLines.length) { clearInterval(_devTypeTimer); return; }
        var line = _devLines[i];
        var el = document.createElement('div');
        el.className = 'dev-line';
        if (line.type === 'log') {
            el.innerHTML = '<span class="dev-log-tag ' + line.tagClass + '">' + line.tag + '</span> <span class="dev-line-text">' + line.text + '</span>';
        } else {
            el.className += ' dev-line-' + line.type;
            el.textContent = line.text || '\u00A0';
        }
        content.appendChild(el);
        i++;
    }, 80);
    var commits = window._devCommitCount || 1;
    var commitCounter = 0; var sanity = 100; var sleep = 8;
    _devStatsTimer = setInterval(function() {
        commitCounter = Math.min(commitCounter + Math.ceil(commits / 30), commits);
        sanity = Math.max(sanity - 4, 0);
        sleep = Math.max(sleep - 0.3, 0);
        document.getElementById('dev-stat-commits').textContent = commitCounter;
        document.getElementById('dev-stat-sleep').textContent = sleep.toFixed(1);
        document.getElementById('dev-stat-sanity').textContent = sanity + '%';
        if (commitCounter >= commits && sanity <= 0) {
            clearInterval(_devStatsTimer);
            document.getElementById('dev-stat-sanity').textContent = 'NaN%';
            document.getElementById('dev-stat-sanity').style.color = '#f87171';
        }
    }, 60);
}

function devModalExit() {
    clearInterval(_devTypeTimer);
    clearInterval(_devStatsTimer);
    var terminal = document.querySelector('.dev-terminal');
    terminal.classList.add('crashing');
    var title = document.querySelector('.dev-terminal-title');
    if (title) title.classList.add('dev-glitch');
    setTimeout(function() {
        document.getElementById('dev-modal').classList.remove('active');
        terminal.classList.remove('crashing');
        if (title) title.classList.remove('dev-glitch');
    }, 650);
}
</script>
